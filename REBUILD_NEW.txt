1. RUN THEN REPLACE ACTUAL FOLDER FOR GIT

npm run build




2. RUN BELOW TO UPDATE FINAL

# 1. Stage all changes
git add .

# 2. Ask for commit message
$msg = Read-Host "Commit message"

if (-not $msg) {
    Write-Host "Commit aborted: empty message" -ForegroundColor Red
    exit 1
}

# 3. Commit changes
git commit -m "$msg"

# 4. Push to GitHub
# This command automatically triggers Vercel to build and deploy!
$branch = git rev-parse --abbrev-ref HEAD
git push origin $branch

Write-Host "Successfully pushed to GitHub. Vercel is now building your update!" -ForegroundColor Green
